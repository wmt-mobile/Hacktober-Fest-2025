/* 2. Automate form submission  
    - Handle scenarios where the success message appears randomly after 1–5 seconds.  
    - Document with clear steps to reproduce, actual vs expected results, and screenshots.*/ 

package automationTests;

import io.github.bonigarcia.wdm.WebDriverManager;
import org.openqa.selenium.*;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.*;
import org.testng.Assert;
import org.testng.annotations.*;

import java.io.File;
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.StandardCopyOption;
import java.time.Duration;

public class FormSubmissionTest {

    WebDriver driver;
    WebDriverWait wait;

    @BeforeClass
    public void setup() {
        WebDriverManager.chromedriver().setup();
        driver = new ChromeDriver();
        driver.manage().window().maximize();
        wait = new WebDriverWait(driver, Duration.ofSeconds(7));
    }

    @Test
    public void testFormSubmission() {
        System.out.println("=== TEST: Automate Form Submission with Random Delay ===");

        try {
            // Step 1: Navigate to form
            driver.get("https://www.techlistic.com/p/selenium-practice-form.html");
            System.out.println("Step 1: Navigated to form page ");

            // Step 2: Fill form fields
            driver.findElement(By.name("firstname")).sendKeys("John");
            driver.findElement(By.name("lastname")).sendKeys("Doe");
            driver.findElement(By.id("sex-0")).click(); // Male
            driver.findElement(By.id("exp-2")).click(); // Experience
            driver.findElement(By.id("datepicker")).sendKeys("10/10/2025");
            driver.findElement(By.id("profession-1")).click(); // Automation Tester
            driver.findElement(By.id("tool-2")).click(); // Selenium Webdriver
            driver.findElement(By.id("continents")).sendKeys("Europe");
            driver.findElement(By.id("submit")).click();
            System.out.println("Step 2: Form submitted");

            // Step 3: Wait for success message (simulate random delay)
            try {
                Thread.sleep((int) (Math.random() * 4000 + 1000)); // 1–5 sec delay simulation
                WebElement successMsg = wait.until(
                        ExpectedConditions.visibilityOfElementLocated(By.tagName("h1"))
                );

                // Capture screenshot
                takeScreenshot("FormSuccess.png");

                // Verify result (assume success message visible)
                Assert.assertTrue(successMsg.isDisplayed(), "Success message not displayed!");

                // Print test documentation
                System.out.println("\n=== TEST DOCUMENTATION ===");
                System.out.println("Steps to Reproduce:");
                System.out.println("1. Open TechListic Practice Form");
                System.out.println("2. Fill valid details");
                System.out.println("3. Click Submit button");
                System.out.println("\nExpected Result:");
                System.out.println("Success message should appear after 1–5 seconds.");
                System.out.println("\nActual Result:");
                System.out.println("Message displayed: " + successMsg.getText());
                System.out.println("Screenshot: FormSuccess.png");
                System.out.println("===========================\n");

            } catch (TimeoutException te) {
                takeScreenshot("FormFailure.png");
                Assert.fail("Success message not found within expected time!");
            }

        } catch (Exception e) {
            takeScreenshot("FormError.png");
            Assert.fail("Test Failed: " + e.getMessage());
        }
    }

    @AfterClass
    public void tearDown() {
        driver.quit();
    }

    private void takeScreenshot(String fileName) {
        File src = ((TakesScreenshot) driver).getScreenshotAs(OutputType.FILE);
        try {
            Files.copy(src.toPath(), new File("./" + fileName).toPath(), StandardCopyOption.REPLACE_EXISTING);
            System.out.println("Screenshot saved: " + fileName);
        } catch (IOException e) {
            System.out.println("Failed to save screenshot: " + e.getMessage());
        }
    }
}
