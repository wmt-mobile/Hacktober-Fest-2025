/*1. Verify validation errors  
    - Check for empty fields, invalid email format, and short password.  
    - Write a generic reusable function that accepts field locators and expected error messages. */

import org.openqa.selenium.*;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import org.openqa.selenium.support.ui.*;
import io.github.bonigarcia.wdm.WebDriverManager;
import org.testng.Assert;

import java.time.Duration;

public class BrowserStackLoginTest {
    public static void main(String[] args) {
        WebDriverManager.chromedriver().setup();
        ChromeOptions options = new ChromeOptions();
        options.addArguments("--headless"); // Run in headless mode
        WebDriver driver = new ChromeDriver(options);

        try {
            driver.get("https://www.browserstack.com/users/sign_in");

            // Test 1: Check for empty fields
            driver.findElement(By.id("user_email_login")).clear();
            driver.findElement(By.id("user_password")).clear();
            driver.findElement(By.name("commit")).click();
            WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));
            wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("user_email_login")));
            wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("user_password")));

            // Test 2: Invalid email format
            driver.findElement(By.id("user_email_login")).sendKeys("invalid-email");
            driver.findElement(By.id("user_password")).sendKeys("ValidPass123");
            driver.findElement(By.name("commit")).click();
            wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("user_email_login")));

            // Test 3: Short password
            driver.findElement(By.id("user_email_login")).clear();
            driver.findElement(By.id("user_email_login")).sendKeys("valid@example.com");
            driver.findElement(By.id("user_password")).clear();
            driver.findElement(By.id("user_password")).sendKeys("short");
            driver.findElement(By.name("commit")).click();
            wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("user_password")));

            // Test 4: Valid input
            driver.findElement(By.id("user_email_login")).clear();
            driver.findElement(By.id("user_email_login")).sendKeys("valid@example.com");
            driver.findElement(By.id("user_password")).clear();
            driver.findElement(By.id("user_password")).sendKeys("ValidPass123");
            driver.findElement(By.name("commit")).click();
            wait.until(ExpectedConditions.urlContains("dashboard"));

            // Verify successful login
            Assert.assertTrue(driver.getCurrentUrl().contains("dashboard"));

        } finally {
            driver.quit();
        }
    }
}
